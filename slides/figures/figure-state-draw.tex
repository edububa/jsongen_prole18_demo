\newcommand\irregularcircle[2]{% radius, irregularity
  +(0:{(#1)+rand*(#2)})
  \foreach \a in {10,20,...,350}{
    -- +(\a:{(#1)+rand*(#2)})
  } -- cycle
}

  \coordinate (c) at (0,0);
  \coordinate (d) at (3,0);

  \draw[red,rounded corners=1mm] (d) \irregularcircle{1cm}{1mm};
